#Add a new column in the data frame for AgeGroup. Values in this column will be Child(age >0 and age <=14), 
#Adult(age>14 and age<=60), Elder(age>60) and _NaN(means value is missing in csv file)
for i in range(0,len(titanic_data)):
    if (titanic_data.iloc[i]['Age'] <= 14):
        titanic_data.loc[i,"AgeGroup"] = 'Child'
    elif ((titanic_data.iloc[i]['Age'] > 14) & (titanic_data.iloc[i]['Age'] <= 60)):
        titanic_data.loc[i,"AgeGroup"] = 'Adult'
    elif (titanic_data.iloc[i]['Age'] > 60):
        titanic_data.loc[i,"AgeGroup"] = 'Senior'
    else:
        titanic_data.loc[i,"AgeGroup"] = '_NaN'
